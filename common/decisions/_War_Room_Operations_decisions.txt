war_room_operation_category_1 = {
	operation_1_ai_assign_generals = {
		visible = {
			is_ai = yes
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { amount = 4 }
			has_less_amount_of_war_room_operation_assigned_generals = yes
		}
		available = {
			any_general_is_not_assigned_to_any_war_room_operation = yes
		}

		fixed_random_seed = no
		days_re_enable = 10

		complete_effect = {
			set_temp_variable = { operation_id = 1 }
			assign_random_general_to_war_room_operation = yes
		}

		ai_will_do = {
			base = 100
		}
	}

	operation_test_decision = {
		visible = {
			is_ai = yes # will still show up when active
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			is_war_room_operation_decision_visible = yes
		}
		available = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			is_war_room_operation_decision_available = yes
		}

		remove_trigger = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			is_war_room_operation_decision_removing = yes
		}
		cancel_trigger = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			is_war_room_operation_decision_canceling = yes
		}

		custom_cost_trigger = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			can_afford_war_room_operation_decision_cost = yes
		}
		custom_cost_text = war_room_operation_decision_custom_cost_text

		fire_only_once = yes
		fixed_random_seed = no
		days_remove = 10

		complete_effect = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			execute_complete_effect_war_room_operation_decision = yes
		}
		remove_effect = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			execute_remove_effect_war_room_operation_decision = yes
		}
		cancel_effect = {
			set_temp_variable = { operation_id = 1 }
			set_temp_variable = { decision_id = token:operation_test_decision }
			execute_cancel_effect_war_room_operation_decision = yes
		}

		ai_will_do = {
			base = 100
		}
	}
}
