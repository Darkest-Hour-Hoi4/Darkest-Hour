# usage:
## GER_SS_spawn_division = yes
### used in STATE scope
GER_SS_spawn_division = {
	if = {
		limit = {
			PREV = {
				has_variable = ss_division_name
				has_variable = ss_division_type
			}
		}
		custom_effect_tooltip = GER_SS_SPAWN_DIVISION_TT
		hidden_effect = {
			PREV = {
				set_temp_variable = { ss_division_type_temp = ss_division_type }
				if = {
					limit = {
						NOT = { has_template = "SS-Infanterie-Division" }
						OR = {
							GER_SS_is_division_type_infantry = yes
							GER_SS_is_division_type_mountaineers = yes
							GER_SS_is_division_type_motorized = yes
							GER_SS_is_division_type_cavalry = yes
						}
					}
					load_oob = "GER_SS_Waffen_Inf"
				}
				if = {
					limit = {
						NOT = { has_template = "SS-Panzergrenadier-Division" }
						OR = {
							GER_SS_is_division_type_mechanized = yes
							GER_SS_is_division_type_panzer = yes
						}
					}
					load_oob = "GER_SS_Waffen_Mech"
				}
				if = {
					limit = {
						NOT = { has_template = "SS-Fallschirm-Division" }
						OR = {
							GER_SS_is_division_type_marines = yes
							GER_SS_is_division_type_paratroopers = yes
						}
					}
					load_oob = "GER_SS_Waffen_SpecOps"
				}
			}
			meta_effect = {
				text = {
					create_unit = {
						division = "name = \"[NAME]\" division_template = \"[TYPE]\" start_experience_factor = 0.8 start_equipment_factor = 0.5"
						owner = PREV
					}
				}
				NAME = "[Prev.GetSSDivisionName]"
				TYPE = "[Prev.GetSSDivisionType]"
			}
		}
		custom_effect_tooltip = GER_SS_SPAWN_DIVISION_TT_3
		PREV = {
			hidden_effect = {
				country_event = { id = GER_SS_Recruitment_Name_Explanation_Events.1 }
			}
		}
	}
	else = {
		custom_effect_tooltip = GER_SS_SPAWN_DIVISION_TT_2
	}
}

# usage:
## GER_SS_division_clear_variables = yes
GER_SS_division_clear_variables = {
	if = {
		limit = {
			has_variable = ss_division_name
		}
		add_to_array = { ss_division_names_used = ss_division_name }
		clear_variable = ss_division_name
	}
	if = {
		limit = {
			has_variable = ss_division_type
		}
		clear_variable = ss_division_type
	}
	clr_country_flag = GER_SS_recruitment_decision_active
}
